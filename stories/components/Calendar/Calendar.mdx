# Calendar

The Calendar component provides a visual interface for navigating and selecting dates in a calendar format.

## Features

- Monthly calendar view
- Month navigation
- Date selection via click
- Selected date state
- Responsive design
- CSS class customization

## Usage

```tsx
import { Calendar } from "@your-package/components";

const MyComponent = () => {
  const [selectedDate, setSelectedDate] = useState<Date | null>(null);

  return (
    <div className="space-y-4">
      <Calendar onDateSelect={(date) => setSelectedDate(date)} />
      {selectedDate && (
        <div className="text-gray-800">
          Selected Date: {selectedDate.toLocaleDateString()}
        </div>
      )}
    </div>
  );
};
```
