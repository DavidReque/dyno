# Table

The Table component is a UI element that displays data in a tabular format with rows and columns.

## Features

- Customizable columns
- Customizable data
- Customizable cells
- Customizable CSS classes
- Paginated and selectable versions

## Usage

```tsx
import { Table, Column, PaginatedTable, SelectableTable } from "@your-package/components";

// Define columns
const columns: Column<Person>[] = [
  {
    header: "Name",
    accessor: "name",
  },
  {
    header: "Age",
    accessor: "age",
  },
  {
    header: "Email",
    accessor: "email",
  },
];

// Define data
const data: Person[] = [
  { name: "John Doe", age: 28, email: "john.doe@example.com" },
  { name: "Jane Smith", age: 34, email: "jane.smith@example.com" },
  { name: "Bob Johnson", age: 42, email: "bob.johnson@example.com" },
];

// Basic table
<Table columns={columns} data={data} />

// Table with custom cells
<Table
  columns={[
    {
      header: "Name",
      accessor: "name",
    },
    {
      header: "Age",
      accessor: "age",
      cell: (row: Person) => (
        <span className="font-bold">{row.age} years</span>
      ),
    },
    {
      header: "Email",
      accessor: "email",
    },
  ]}
  data={data}
/>

// Paginated table
<PaginatedTable<Person> columns={columns} data={data} pageSize={3} />

// Selectable table
<SelectableTable
  columns={columns}
  data={data}
  onSelectionChange={(selected) => console.log("Selected rows:", selected)}
/>
```
